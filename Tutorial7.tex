\subsection*{Kruskal's algorithm}

\begin{enumerate}
    \item step1 merge sort: $O(|E|\log{|E|})$
    \item step2 $O(|E|)$
    \item total:
        \begin{align*}
            O(|E|\log{|E|})+O(|E|)&=O(|E|\log{|E|})\\
            &=O(|V|^2\log{|V|})
        \end{align*}
\end{enumerate}
